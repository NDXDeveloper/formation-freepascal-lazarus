  {$IFDEF WINDOWS}

  Result := ((Length(APath) >= 3) and (APath[2] = ':') and (APath[3] = '\')) or

  Result := (Length(APath) > 0) and (APath[1] = '/');
  {$ENDIF}

  Result := CreateRelativePath(APath, ABase);
