      EvictLRU;

    if Result then
    begin

      begin
        Result := False;
        FLock.EndRead;
        FLock.BeginWrite;
        try
          FItems.Remove(Key);
        finally
          FLock.EndWrite;
          FLock.BeginRead;
        end;
      end
